# Tether v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a self-hosted Claude Code mobile companion (PWA) + cross-attach terminal client that share the same server-owned CC sessions over an end-to-end encrypted transport.

**Architecture:** Two-tier — a Node.js single-binary `tether-cli` runs on the user's server, spawns CC inside `dtach` sockets for terminal multiplexing, watches CC's JSONL output, and exposes a WebTransport (with WS-over-TLS fallback) channel to a React PWA. All payloads are E2E encrypted with libsodium box + AES-256-GCM (crypto code copied verbatim from happy-cli). Web Push API delivers wake-up notifications.

**Tech Stack:** TypeScript 5.6+, Node.js 20+, vitest, dtach (binary dep), `ws`, `@fails-components/webtransport`, `tweetnacl`, `zod`, `web-push`, `chokidar`, `commander`, `pino`, `qrcode`. PWA: Vite + React 18 + React Router 6, `libsodium-wrappers`, `idb`, `vite-plugin-pwa`.

**Source we fork from:** `slopus/happy` v1.1.8 at `/tmp/happy-analysis/happy/packages/happy-cli/src/` (MIT). File-by-file mapping in spec §8.

**Spec source of truth:** `docs/specs/2026-04-25-tether-v0.1-design.md`. If this plan and the spec disagree, spec wins — flag and ask.

## Issues found while drafting

1. **AES algorithm naming conflict.** Spec §7 says "AES-256-CTR" but happy-cli's `api/encryption.ts:122` uses **AES-256-GCM** for the `dataKey` variant (which the spec also tells us to copy verbatim). I default to GCM in this plan because we are copying that file. Confirm before Week 2 starts.
2. **dtach raw-mode TUI under multiple attaches** is open question §14.3. If it doesn't work, the entire spec changes (we'd need node-pty + custom fan-out). The plan puts a verification PoC at the very start of Week 1 (Task 1.0). Do not skip.
3. **Special-commands handling under dtach**: the `/clear`, `/resume`, `/compact` flow assumes happy's hook-driven session-id refresh works whether CC is running inside dtach or not. Spec assumes yes; we test in Task 3.6.
4. **Web Push on iOS Safari** is gated on PWA being installed via "Add to Home Screen". Documented in week 6 docs but the PWA must show install instructions on first launch — Task 4.2.

```typescript
// packages/tether-cli/src/store/homedir.ts
import { homedir } from 'node:os';
import { join } from 'node:path';
import { mkdirSync, chmodSync } from 'node:fs';

export interface HomePaths {
    root: string;
    secret: string;
    keypair: string;
    vapid: string;
    tokens: string;
    subscriptions: string;
    sockets: string;
    log: string;
}

export function homePaths(root?: string): HomePaths {
    const base = root ?? join(homedir(), '.tether');
    return {
        root: base,
        secret: join(base, 'secret'),
        keypair: join(base, 'keypair.json'),
        vapid: join(base, 'vapid.json'),
        tokens: join(base, 'tokens.json'),
        subscriptions: join(base, 'subscriptions.json'),
        sockets: join(base, 'sockets'),
        log: join(base, 'log')
    };
}

export function ensureHomeDir(paths: HomePaths): void {
    mkdirSync(paths.root, { recursive: true });
    mkdirSync(paths.sockets, { recursive: true });
    mkdirSync(paths.log, { recursive: true });
    if (process.platform !== 'win32') {
        chmodSync(paths.root, 0o700);
    }
}
```

```typescript
import { homedir } from 'node:os';

export function getProjectPath(workingDirectory: string): string {
```

```typescript
// packages/tether-cli/src/session/dtach.ts
import { spawn, spawnSync, type ChildProcess } from 'node:child_process';
import { existsSync } from 'node:fs';
import { log } from '../log.js';

export interface DtachOptions {
    socketPath: string;
    command: string;
    args?: string[];
    cwd?: string;
    env?: NodeJS.ProcessEnv;
}

export interface DtachHandle {
    socketPath: string;
    pid: number;
    child: ChildProcess;
}

export function isDtachAvailable(): boolean {
    const r = spawnSync('dtach', ['-v'], { stdio: 'ignore' });
    return r.status === 0;
}

export async function spawnDetached(opts: DtachOptions): Promise<DtachHandle> {
    if (existsSync(opts.socketPath)) {
        throw new Error(`socket already exists: ${opts.socketPath}`);
    }
    // -n: spawn detached, no attach
    // -E: ignore detach key
    // -z: disable suspend key
    const args = ['-n', opts.socketPath, '-E', '-z', opts.command, ...(opts.args ?? [])];
    const child = spawn('dtach', args, {
        cwd: opts.cwd,
        env: opts.env,
        stdio: 'ignore',
        detached: true
    });
    child.unref();
    log.debug({ socket: opts.socketPath, cmd: opts.command, pid: child.pid }, 'dtach spawned');

    // Poll for socket file (dtach creates it after fork)
    const deadline = Date.now() + 3000;
    while (!existsSync(opts.socketPath)) {
        if (Date.now() > deadline) throw new Error('timed out waiting for dtach socket');
        await new Promise((r) => setTimeout(r, 25));
    }
    return { socketPath: opts.socketPath, pid: child.pid ?? -1, child };
}

export async function attachInput(socketPath: string, text: string): Promise<void> {
    // dtach -p reads from stdin and forwards to the attached process
    return new Promise((resolve, reject) => {
        const p = spawn('dtach', ['-p', socketPath], { stdio: ['pipe', 'ignore', 'inherit'] });
        p.on('error', reject);
        p.on('exit', (code) => code === 0 ? resolve() : reject(new Error(`dtach -p exit ${code}`)));
        p.stdin.end(text);
    });
}

export function killSession(handle: DtachHandle): void {
    try {
        if (handle.child.pid) process.kill(-handle.child.pid, 'SIGTERM');
    } catch (err) {
        log.debug({ err }, 'killSession ignored error');
    }
}
```

```typescript
// packages/tether-cli/src/cc/jsonl.ts
import chokidar from 'chokidar';
import { open, type FileHandle } from 'node:fs/promises';
import { log } from '../log.js';

export interface WatchOptions {
    onLine: (parsed: unknown, raw: string) => void;
    onError?: (err: Error) => void;
}

export interface JsonlWatcher {
    close: () => Promise<void>;
}

export async function watchJsonl(path: string, opts: WatchOptions): Promise<JsonlWatcher> {
    let offset = 0;
    let buf = '';
    let fh: FileHandle | undefined;
    let closed = false;

    async function read() {
        if (closed) return;
        if (!fh) {
            try { fh = await open(path, 'r'); } catch (err) {
                opts.onError?.(err as Error);
                return;
            }
        }
        const stat = await fh.stat();
        if (stat.size < offset) {
            // file truncated/rotated — restart
            offset = 0;
            buf = '';
        }
        if (stat.size === offset) return;
        const length = stat.size - offset;
        const out = Buffer.alloc(length);
        const { bytesRead } = await fh.read(out, 0, length, offset);
        offset += bytesRead;
        buf += out.toString('utf8', 0, bytesRead);

        let nl: number;
        while ((nl = buf.indexOf('\n')) !== -1) {
            const line = buf.slice(0, nl);
            buf = buf.slice(nl + 1);
            if (!line.trim()) continue;
            try {
                opts.onLine(JSON.parse(line), line);
            } catch (err) {
                opts.onError?.(err as Error);
            }
        }
    }

    const w = chokidar.watch(path, { persistent: true, awaitWriteFinish: false });
    w.on('add', () => { void read(); });
    w.on('change', () => { void read(); });
    w.on('error', (err) => opts.onError?.(err as Error));

    // initial read for already-present content
    await read();

    return {
        close: async () => {
            closed = true;
            await w.close();
            if (fh) await fh.close();
        }
    };
}
```

```typescript
import { generateHookSettingsFile } from '../cc/hookSettings.js';

    let resolvedSessionId = sessionId;

        args: ['--session-id', sessionId, '--settings', settings.path],

    const jsonlPath = join(getProjectPath(projectDir), `${resolvedSessionId}.jsonl`);

        onError: (err) => log.warn({ err: err.message }, 'jsonl-error')

        hookServer.stop();
        settings.cleanup();
        killSession(dtachHandle);
        process.exit(0);

    process.on('SIGINT', shutdown);
```

```typescript
// packages/tether-wire/src/envelope.ts
import { z } from 'zod';

export const MessageTypeSchema = z.enum([
    'pair-request',
    'pair-response',
    'rpc-call',
    'rpc-response',
    'session-update',
    'session-state',
    'session-list-update',
    'session-spawn',
    'session-input',
    'session-attach',
    'session-detach',
    'session-kill',
    'push-token-register'
]);
export type MessageType = z.infer<typeof MessageTypeSchema>;

export const EnvelopeSchema = z.object({
    v: z.literal(1),
    id: z.string().min(1),
    type: MessageTypeSchema,
    encrypted: z.boolean(),
    payload: z.instanceof(Uint8Array),
    timestamp: z.number().int().nonnegative(),
    userId: z.string().optional(),
    machineId: z.string().optional()
});
export type Envelope = z.infer<typeof EnvelopeSchema>;

// Wire format: 4-byte BE length prefix + JSON header + 4-byte BE payload length + payload bytes.
// Header includes everything except payload bytes (which are binary so kept separate).
export function encodeEnvelope(env: Envelope): Uint8Array {
    EnvelopeSchema.parse(env);
    const header = JSON.stringify({
        v: env.v, id: env.id, type: env.type, encrypted: env.encrypted,
        timestamp: env.timestamp, userId: env.userId, machineId: env.machineId
    });
    const headerBytes = new TextEncoder().encode(header);
    const out = new Uint8Array(4 + headerBytes.length + 4 + env.payload.length);
    const dv = new DataView(out.buffer);
    dv.setUint32(0, headerBytes.length, false);
    out.set(headerBytes, 4);
    dv.setUint32(4 + headerBytes.length, env.payload.length, false);
    out.set(env.payload, 4 + headerBytes.length + 4);
    return out;
}

export function decodeEnvelope(bytes: Uint8Array): Envelope {
    const dv = new DataView(bytes.buffer, bytes.byteOffset, bytes.byteLength);
    const headerLen = dv.getUint32(0, false);
    const header = JSON.parse(new TextDecoder().decode(bytes.subarray(4, 4 + headerLen)));
    const payloadLen = dv.getUint32(4 + headerLen, false);
    const payload = bytes.subarray(4 + headerLen + 4, 4 + headerLen + 4 + payloadLen);
    return EnvelopeSchema.parse({ ...header, payload });
}
```

```typescript
// packages/tether-cli/src/crypto/keystore.ts
import { existsSync, readFileSync, writeFileSync, chmodSync } from 'node:fs';
import { libsodiumPublicKeyFromSecretKey, getRandomBytes, encodeBase64, decodeBase64 } from './encryption.js';
import type { HomePaths } from '../store/homedir.js';

export interface ServerKeypair {
    secretSeed: Uint8Array;
    publicKey: Uint8Array;
}

export async function ensureServerKeypair(paths: HomePaths): Promise<ServerKeypair> {
    if (existsSync(paths.keypair)) {
        const j = JSON.parse(readFileSync(paths.keypair, 'utf8'));
        const seed = decodeBase64(j.secretSeed);
        return { secretSeed: seed, publicKey: libsodiumPublicKeyFromSecretKey(seed) };
    }
    const seed = getRandomBytes(32);
    const pub = libsodiumPublicKeyFromSecretKey(seed);
    writeFileSync(paths.keypair, JSON.stringify({
        secretSeed: encodeBase64(seed),
        publicKey: encodeBase64(pub)
    }), 'utf8');
    if (process.platform !== 'win32') chmodSync(paths.keypair, 0o600);
    return { secretSeed: seed, publicKey: pub };
}
```

```typescript
// packages/tether-cli/src/transport/websocket.ts
import { WebSocketServer, type WebSocket } from 'ws';
import { createServer as createHttpServer, type Server } from 'node:http';
import { randomUUID } from 'node:crypto';
import { encodeEnvelope, decodeEnvelope, type Envelope } from '@tether/wire';
import type { MessageStream, TransportServer } from './protocol.js';
import { log } from '../log.js';

export interface WebSocketTransportOptions {
    host: string;
    port: number;
    httpServer?: Server;        // when sharing with the static asset server
}

export async function startWebSocketServer(opts: WebSocketTransportOptions): Promise<TransportServer> {
    const httpServer = opts.httpServer ?? createHttpServer();
    const wss = new WebSocketServer({ server: httpServer, path: '/wt' });
    let onConn: ((s: MessageStream) => void) | undefined;

    wss.on('connection', (ws: WebSocket, req) => {
        const stream = wrap(ws, `ws:${req.socket.remoteAddress}:${req.socket.remotePort}`);
        onConn?.(stream);
    });

    return {
        onConnection: (h) => { onConn = h; },
        listen: () => new Promise((resolve, reject) => {
            httpServer.once('error', reject);
            httpServer.listen(opts.port, opts.host, () => {
                const addr = httpServer.address();
                if (!addr || typeof addr === 'string') return reject(new Error('no address'));
                resolve({ port: addr.port });
            });
        }),
        close: () => new Promise((resolve) => {
            wss.close(() => httpServer.close(() => resolve()));
        })
    };
}

function wrap(ws: WebSocket, remote: string): MessageStream {
    let onMsg: ((e: Envelope) => void) | undefined;
    let onClose: (() => void) | undefined;
    ws.binaryType = 'nodebuffer';
    ws.on('message', (data: Buffer) => {
        try {
            onMsg?.(decodeEnvelope(new Uint8Array(data)));
        } catch (err) {
            log.warn({ err }, 'invalid envelope');
        }
    });
    ws.on('close', () => onClose?.());
    return {
        id: randomUUID(),
        remote,
        send: (env) => new Promise((resolve, reject) => {
            ws.send(encodeEnvelope(env), (err) => err ? reject(err) : resolve());
        }),
        onMessage: (h) => { onMsg = h; },
        onClose: (h) => { onClose = h; },
        close: () => ws.close()
    };
}
```

```typescript
// packages/tether-cli/src/transport/webtransport.ts
import { Http3Server } from '@fails-components/webtransport';
import { randomUUID } from 'node:crypto';
import { encodeEnvelope, decodeEnvelope, type Envelope } from '@tether/wire';
import type { MessageStream, TransportServer } from './protocol.js';
import { log } from '../log.js';

export interface WebTransportOptions {
    host: string;
    port: number;
    cert: string;       // PEM
    key: string;        // PEM
}

export async function startWebTransportServer(opts: WebTransportOptions): Promise<TransportServer> {
    const server = new Http3Server({
        host: opts.host,
        port: opts.port,
        cert: opts.cert,
        privKey: opts.key,
        secret: 'tether-wt'
    });
    let onConn: ((s: MessageStream) => void) | undefined;
    let listening = false;

    server.startServer();

    (async () => {
        const sessionStream = server.sessionStream('/wt');
        const reader = sessionStream.getReader();
        for (;;) {
            const { value: session, done } = await reader.read();
            if (done) break;
            await session.ready;
            const stream = wrapWtSession(session);
            onConn?.(stream);
        }
    })().catch((err) => log.warn({ err }, 'webtransport accept loop ended'));

    return {
        onConnection: (h) => { onConn = h; },
        listen: async () => {
            if (!listening) {
                await server.ready;
                listening = true;
            }
            return { port: opts.port };
        },
        close: async () => {
            await server.stopServer();
        }
    };
}

function wrapWtSession(session: any): MessageStream {
    let onMsg: ((e: Envelope) => void) | undefined;
    let onClose: (() => void) | undefined;

    (async () => {
        const reader = session.incomingBidirectionalStreams.getReader();
        for (;;) {
            const { value: bidi, done } = await reader.read();
            if (done) break;
            (async () => {
                const r = bidi.readable.getReader();
                const chunks: Uint8Array[] = [];
                for (;;) {
                    const { value, done } = await r.read();
                    if (done) break;
                    chunks.push(value);
                }
                const total = chunks.reduce((n, c) => n + c.length, 0);
                const buf = new Uint8Array(total);
                let off = 0;
                for (const c of chunks) { buf.set(c, off); off += c.length; }
                try { onMsg?.(decodeEnvelope(buf)); } catch (err) { log.warn({ err }, 'wt decode failed'); }
            })().catch((err) => log.warn({ err }, 'wt stream read failed'));
        }
    })().catch(() => onClose?.());

    session.closed.then(() => onClose?.()).catch(() => onClose?.());

    return {
        id: randomUUID(),
        remote: 'webtransport',
        send: async (env) => {
            const bidi = await session.createBidirectionalStream();
            const w = bidi.writable.getWriter();
            await w.write(encodeEnvelope(env));
            await w.close();
        },
        onMessage: (h) => { onMsg = h; },
        onClose: (h) => { onClose = h; },
        close: () => session.close()
    };
}
```

```typescript
// packages/tether-cli/src/transport/server.ts
import { createServer as createHttpServer, type Server as HttpServer } from 'node:http';
import { startWebSocketServer } from './websocket.js';
import { startWebTransportServer } from './webtransport.js';
import type { MessageStream, TransportServer } from './protocol.js';

export interface CombinedHostOptions {
    host: string;
    port: number;
    cert?: string;          // when provided, also boots WT
    key?: string;
    onConnection: (stream: MessageStream) => void;
    httpHandler?: (req: any, res: any) => void; // static asset server
}

export interface CombinedHost {
    httpServer: HttpServer;
    ws: TransportServer;
    wt?: TransportServer;
    listen: () => Promise<{ port: number }>;
    close: () => Promise<void>;
}

export async function startCombinedHost(opts: CombinedHostOptions): Promise<CombinedHost> {
    const httpServer = createHttpServer((req, res) => opts.httpHandler?.(req, res) ?? res.writeHead(404).end());
    const ws = await startWebSocketServer({ host: opts.host, port: opts.port, httpServer });
    ws.onConnection(opts.onConnection);

    let wt: TransportServer | undefined;
    if (opts.cert && opts.key) {
        try {
            wt = await startWebTransportServer({ host: opts.host, port: opts.port, cert: opts.cert, key: opts.key });
            wt.onConnection(opts.onConnection);
        } catch (err) {
            // WT failure is non-fatal — clients will fall back to WS
        }
    }

    return {
        httpServer, ws, wt,
        listen: async () => {
            const r = await ws.listen();
            if (wt) await wt.listen();
            return r;
        },
        close: async () => {
            await ws.close();
            if (wt) await wt.close();
        }
    };
}
```

```typescript
// packages/tether-cli/src/auth/token.ts
import { randomUUID } from 'node:crypto';
import { existsSync, readFileSync, writeFileSync } from 'node:fs';
import type { HomePaths } from '../store/homedir.js';

interface BearerRecord {
    token: string;
    clientPubKey: string;
    issuedAt: number;
    expiresAt: number;
}

interface OneTimeRecord {
    token: string;
    expiresAt: number;
}

interface TokenFile {
    bearers: BearerRecord[];
    oneTimes: OneTimeRecord[];
}

export class TokenStore {
    private file: TokenFile;

    constructor(private paths: HomePaths) {
        this.file = existsSync(paths.tokens)
            ? JSON.parse(readFileSync(paths.tokens, 'utf8'))
            : { bearers: [], oneTimes: [] };
    }

    private save() {
        writeFileSync(this.paths.tokens, JSON.stringify(this.file), 'utf8');
    }

    issueOneTimeToken(ttlMs = 5 * 60_000): string {
        const token = randomUUID();
        this.file.oneTimes.push({ token, expiresAt: Date.now() + ttlMs });
        this.save();
        return token;
    }

    consumeOneTimeToken(token: string): boolean {
        const idx = this.file.oneTimes.findIndex((o) => o.token === token && o.expiresAt > Date.now());
        if (idx === -1) return false;
        this.file.oneTimes.splice(idx, 1);
        this.save();
        return true;
    }

    issueBearer(opts: { clientPubKey: string; expiresInMs: number }): string {
        const rec: BearerRecord = {
            token: randomUUID(),
            clientPubKey: opts.clientPubKey,
            issuedAt: Date.now(),
            expiresAt: Date.now() + opts.expiresInMs
        };
        this.file.bearers.push(rec);
        this.save();
        return rec.token;
    }

    verifyBearer(token: string): BearerRecord | null {
        const r = this.file.bearers.find((b) => b.token === token);
        if (!r) return null;
        if (r.expiresAt <= Date.now()) return null;
        return r;
    }

    revokeBearer(token: string): void {
        this.file.bearers = this.file.bearers.filter((b) => b.token !== token);
        this.save();
    }
}
```

```typescript
// packages/tether-cli/src/auth/pair.ts
import { encodeBase64 } from '../crypto/encryption.js';
import type { ServerKeypair } from '../crypto/keystore.js';
import type { TokenStore } from './token.js';
import type { PairRequest, PairResponse } from '@tether/wire';

const BEARER_TTL_MS = 30 * 24 * 60 * 60 * 1000;

export interface PairerOptions {
    tokens: TokenStore;
    server: ServerKeypair;
    machineId: string;
}

export type Result<T> = { ok: true; value: T } | { ok: false; error: { code: string; message: string } };

export class Pairer {
    constructor(private opts: PairerOptions) {}

    handlePairRequest(req: PairRequest): Result<PairResponse> {
        if (!this.opts.tokens.consumeOneTimeToken(req.oneTimeToken)) {
            return { ok: false, error: { code: 'invalid-ott', message: 'one-time token invalid or expired' } };
        }
        const bearer = this.opts.tokens.issueBearer({
            clientPubKey: req.clientPubKey,
            expiresInMs: BEARER_TTL_MS
        });
        return {
            ok: true,
            value: {
                serverPubKey: encodeBase64(this.opts.server.publicKey),
                bearerToken: bearer,
                expiresAt: Date.now() + BEARER_TTL_MS,
                machineId: this.opts.machineId
            }
        };
    }
}

export function formatPairingUrl(opts: { serverUrl: string; serverPubKey: string; oneTimeToken: string }): string {
    const u = new URL(opts.serverUrl);
    return `tether://${u.host}?pubkey=${encodeURIComponent(opts.serverPubKey)}&token=${encodeURIComponent(opts.oneTimeToken)}`;
}
```

```typescript
import qrcode from 'qrcode-terminal';
import { randomUUID } from 'node:crypto';
import { existsSync, readFileSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';
import { homePaths, ensureHomeDir } from '../store/homedir.js';
import { ensureServerKeypair } from '../crypto/keystore.js';
import { encodeBase64 } from '../crypto/encryption.js';
import { TokenStore } from '../auth/token.js';
import { Pairer, formatPairingUrl } from '../auth/pair.js';
import { startCombinedHost } from '../transport/server.js';
import { devCert } from '../transport/devCert.js';
import { staticHandler } from '../http/static.js';
import { decodeEnvelope, encodeEnvelope, EnvelopeSchema, PairRequestSchema } from '@tether/wire';
import type { MessageStream } from '../transport/protocol.js';
import { log } from '../log.js';

export interface StartOptions {
    publicUrl: string;
    host?: string;
    port?: number;
}

export async function startCommand(opts: StartOptions): Promise<void> {
    const paths = homePaths();
    ensureHomeDir(paths);

    const server = await ensureServerKeypair(paths);
    const tokens = new TokenStore(paths);

    const machineIdFile = join(paths.root, 'machineId');
    let machineId: string;
    if (existsSync(machineIdFile)) {
        machineId = readFileSync(machineIdFile, 'utf8').trim();
    } else {
        machineId = randomUUID();
        writeFileSync(machineIdFile, machineId, 'utf8');
    }

    const pairer = new Pairer({ tokens, server, machineId });

    const ott = tokens.issueOneTimeToken();
    const pairUrl = formatPairingUrl({
        serverUrl: opts.publicUrl,
        serverPubKey: encodeBase64(server.publicKey),
        oneTimeToken: ott
    });

    const cert = devCert(new URL(opts.publicUrl).hostname);

    const host = await startCombinedHost({
        host: opts.host ?? '0.0.0.0',
        port: opts.port ?? 7777,
        cert: cert.cert,
        key: cert.key,
        httpHandler: staticHandler,
        onConnection: (stream) => attach(stream, pairer)
    });

    const { port } = await host.listen();

    log.info({ port, publicUrl: opts.publicUrl, machineId }, 'tether-cli listening');
    log.info('Scan this QR with the Tether PWA:');
    qrcode.generate(pairUrl, { small: true });
    log.info({ pairUrl }, 'pairing URL');

    process.on('SIGINT', async () => { await host.close(); process.exit(0); });
}

function attach(stream: MessageStream, pairer: Pairer): void {
    stream.onMessage((env) => {
        if (env.type === 'pair-request') {
            const inner = JSON.parse(new TextDecoder().decode(env.payload));
            const req = PairRequestSchema.parse(inner);
            const result = pairer.handlePairRequest(req);
            const respPayload = result.ok
                ? new TextEncoder().encode(JSON.stringify(result.value))
                : new TextEncoder().encode(JSON.stringify({ error: result.error }));
            void stream.send({
                v: 1, id: env.id, type: 'pair-response',
                encrypted: false, payload: respPayload, timestamp: Date.now()
            });
            return;
        }
        // other message types handled in week 3
    });
}
```

```typescript
// packages/tether-cli/src/session/manager.ts
export interface SessionRecord {
    id: string;
    cwd: string;
    sock: string;
    pid: number;
    jsonlPath: string;
    createdAt: number;
    lastActivity?: number;
    attachedClients: Set<string>;
    attachedTerminal: boolean;
}

export interface SessionView {
    id: string;
    cwd: string;
    createdAt: number;
    lastActivity?: number;
    attachedClients: number;
    attachedTerminal: boolean;
}

export class SessionManager {
    private sessions = new Map<string, SessionRecord>();

    register(input: { id: string; cwd: string; sock: string; pid: number; jsonlPath: string }): string {
        const rec: SessionRecord = {
            ...input,
            createdAt: Date.now(),
            attachedClients: new Set(),
            attachedTerminal: false
        };
        this.sessions.set(input.id, rec);
        return input.id;
    }

    unregister(id: string): void { this.sessions.delete(id); }

    get(id: string): SessionRecord | undefined { return this.sessions.get(id); }

    list(): SessionView[] {
        return Array.from(this.sessions.values()).map((r) => ({
            id: r.id, cwd: r.cwd, createdAt: r.createdAt, lastActivity: r.lastActivity,
            attachedClients: r.attachedClients.size, attachedTerminal: r.attachedTerminal
        }));
    }

    attachClient(sessionId: string, clientId: string): void {
        this.sessions.get(sessionId)?.attachedClients.add(clientId);
    }

    detachClient(sessionId: string, clientId: string): void {
        this.sessions.get(sessionId)?.attachedClients.delete(clientId);
    }

    touchActivity(sessionId: string): void {
        const r = this.sessions.get(sessionId);
        if (r) r.lastActivity = Date.now();
    }
}
```

```typescript
// packages/tether-cli/src/session/multiplexer.ts
export interface MuxEvent { seq: number; line: unknown; }
export type Subscriber = (e: MuxEvent) => void;

interface SessionBuffer {
    events: MuxEvent[];
    subscribers: Map<string, Subscriber>;
}

export interface MultiplexerOptions { replayBufferSize: number; }

export class Multiplexer {
    private buffers = new Map<string, SessionBuffer>();

    constructor(private opts: MultiplexerOptions) {}

    private ensure(sessionId: string): SessionBuffer {
        let b = this.buffers.get(sessionId);
        if (!b) { b = { events: [], subscribers: new Map() }; this.buffers.set(sessionId, b); }
        return b;
    }

    emit(sessionId: string, ev: MuxEvent): void {
        const b = this.ensure(sessionId);
        b.events.push(ev);
        if (b.events.length > this.opts.replayBufferSize) b.events.shift();
        for (const s of b.subscribers.values()) {
            try { s(ev); } catch { /* subscriber errors don't break others */ }
        }
    }

    attach(sessionId: string, clientId: string, sub: Subscriber, opts?: { sinceSeq?: number }): void {
        const b = this.ensure(sessionId);
        b.subscribers.set(clientId, sub);
        if (opts?.sinceSeq !== undefined) {
            for (const e of b.events) if (e.seq > opts.sinceSeq) sub(e);
        }
    }

    detach(sessionId: string, clientId: string): void {
        this.buffers.get(sessionId)?.subscribers.delete(clientId);
    }

    drop(sessionId: string): void { this.buffers.delete(sessionId); }
}
```

```typescript
// Inspired by happy v1.1.8 (MIT) — packages/happy-cli/src/claude/claudeLocal.ts.
// Adaptations: spawn CC inside a dtach socket so multiple terminals + the PWA can attach.
import { join } from 'node:path';
import { randomUUID } from 'node:crypto';
import { startHookServer } from './hookServer.js';
import { generateHookSettingsFile } from './hookSettings.js';
import { spawnDetached, type DtachHandle } from '../session/dtach.js';
import { getProjectPath } from './path.js';
import { homePaths, ensureHomeDir } from '../store/homedir.js';
import { log } from '../log.js';

export interface SpawnOptions {
    cwd: string;
    resumeSessionId?: string;
    extraArgs?: string[];
    claudeBinary?: string;
}

export interface SpawnResult {
    sessionId: string;
    sock: string;
    pid: number;
    jsonlPath: string;
    cleanup: () => void;
}

export async function spawnSession(opts: SpawnOptions): Promise<SpawnResult> {
    const paths = homePaths();
    ensureHomeDir(paths);

    const seedId = opts.resumeSessionId ?? randomUUID();
    let resolvedSessionId = seedId;
    let resolveHook: () => void = () => {};
    const firstHook = new Promise<void>((r) => { resolveHook = r; });

    const hookServer = await startHookServer({
        onSessionHook: (id) => {
            log.debug({ id }, 'hook fired');
            resolvedSessionId = id;
            resolveHook();
        }
    });
    const settings = generateHookSettingsFile(hookServer.port);

    const sock = join(paths.sockets, `${seedId}.sock`);
    const args = [
        '--session-id', seedId,
        '--settings', settings.path,
        ...(opts.resumeSessionId ? ['--resume', opts.resumeSessionId] : []),
        ...(opts.extraArgs ?? [])
    ];

    let dtachHandle: DtachHandle;
    try {
        dtachHandle = await spawnDetached({
            socketPath: sock,
            command: opts.claudeBinary ?? 'claude',
            args,
            cwd: opts.cwd
        });
    } catch (err) {
        hookServer.stop();
        settings.cleanup();
        throw err;
    }

    // Wait up to 10s for the hook to fire so we know the JSONL path.
    await Promise.race([
        firstHook,
        new Promise<void>((_, reject) => setTimeout(() => reject(new Error('timed out waiting for SessionStart hook')), 10_000))
    ]).catch((err) => {
        hookServer.stop();
        settings.cleanup();
        throw err;
    });

    const jsonlPath = join(getProjectPath(opts.cwd), `${resolvedSessionId}.jsonl`);

    return {
        sessionId: resolvedSessionId,
        sock,
        pid: dtachHandle.pid,
        jsonlPath,
        cleanup: () => {
            hookServer.stop();
            settings.cleanup();
        }
    };
}
```

```typescript
// packages/tether-cli/src/transport/sessionDispatch.ts
import { TextDecoder, TextEncoder } from 'node:util';
import {
    EnvelopeSchema, SessionInputSchema, SessionAttachSchema, SessionDetachSchema,
    SessionKillSchema, SessionSpawnSchema, type Envelope
} from '@tether/wire';
import type { MessageStream } from './protocol.js';
import { SessionManager } from '../session/manager.js';
import { Multiplexer } from '../session/multiplexer.js';
import { spawnSession } from '../cc/spawn.js';
import { attachInput, killSession } from '../session/dtach.js';
import { watchJsonl, type JsonlWatcher } from '../cc/jsonl.js';
import { log } from '../log.js';
import { TokenStore } from '../auth/token.js';

interface AuthedClient { clientId: string; bearer: string; subscriptions: Set<string>; }

export interface DispatchContext {
    manager: SessionManager;
    mux: Multiplexer;
    tokens: TokenStore;
    watchers: Map<string, JsonlWatcher>;
    sequences: Map<string, number>;
}

export function makeDispatchContext(): DispatchContext {
    return {
        manager: new SessionManager(),
        mux: new Multiplexer({ replayBufferSize: 500 }),
        tokens: undefined as unknown as TokenStore, // injected by start
        watchers: new Map(),
        sequences: new Map()
    };
}

export function attachSessionDispatch(stream: MessageStream, ctx: DispatchContext): void {
    const client: AuthedClient = { clientId: stream.id, bearer: '', subscriptions: new Set() };

    stream.onMessage(async (env) => {
        try {
            EnvelopeSchema.parse(env);
        } catch (err) {
            log.warn({ err }, 'invalid envelope');
            return;
        }

        // pair-request handled elsewhere (start.ts) before this dispatch wraps the stream.
        // Subsequent messages must carry a bearer in payload.bearer (in the inner JSON) until encryption is fully wired in week 6.

        if (env.type === 'session-spawn') {
            const body = SessionSpawnSchema.parse(JSON.parse(new TextDecoder().decode(env.payload)));
            const r = await spawnSession({ cwd: body.cwd, resumeSessionId: body.resumeSessionId });
            ctx.manager.register({ id: r.sessionId, cwd: body.cwd, sock: r.sock, pid: r.pid, jsonlPath: r.jsonlPath });
            ctx.sequences.set(r.sessionId, 0);

            const watcher = await watchJsonl(r.jsonlPath, {
                onLine: (parsed) => {
                    const seq = (ctx.sequences.get(r.sessionId) ?? 0);
                    ctx.sequences.set(r.sessionId, seq + 1);
                    ctx.mux.emit(r.sessionId, { seq, line: parsed });
                    ctx.manager.touchActivity(r.sessionId);
                },
                onError: (err) => log.warn({ err: err.message }, 'jsonl error')
            });
            ctx.watchers.set(r.sessionId, watcher);

            broadcastSessionList(ctx, stream);
            replyOk(stream, env, { sessionId: r.sessionId });
            return;
        }

        if (env.type === 'session-attach') {
            const body = SessionAttachSchema.parse(JSON.parse(new TextDecoder().decode(env.payload)));
            ctx.manager.attachClient(body.sessionId, client.clientId);
            client.subscriptions.add(body.sessionId);
            ctx.mux.attach(body.sessionId, client.clientId, (ev) => {
                void stream.send(makeEnv('session-update', { sessionId: body.sessionId, seq: ev.seq, line: ev.line }, env.id));
            }, { sinceSeq: body.sinceSeq ?? -1 });
            replyOk(stream, env, { ok: true });
            return;
        }

        if (env.type === 'session-detach') {
            const body = SessionDetachSchema.parse(JSON.parse(new TextDecoder().decode(env.payload)));
            ctx.manager.detachClient(body.sessionId, client.clientId);
            ctx.mux.detach(body.sessionId, client.clientId);
            client.subscriptions.delete(body.sessionId);
            replyOk(stream, env, { ok: true });
            return;
        }

        if (env.type === 'session-input') {
            const body = SessionInputSchema.parse(JSON.parse(new TextDecoder().decode(env.payload)));
            const rec = ctx.manager.get(body.sessionId);
            if (!rec) return replyErr(stream, env, 'no-such-session');
            await attachInput(rec.sock, body.text);
            replyOk(stream, env, { ok: true });
            return;
        }

        if (env.type === 'session-kill') {
            const body = SessionKillSchema.parse(JSON.parse(new TextDecoder().decode(env.payload)));
            const rec = ctx.manager.get(body.sessionId);
            if (!rec) return replyErr(stream, env, 'no-such-session');
            // SIGTERM the dtach + cc subtree
            try { process.kill(-rec.pid, 'SIGTERM'); } catch {}
            const w = ctx.watchers.get(body.sessionId);
            if (w) await w.close();
            ctx.watchers.delete(body.sessionId);
            ctx.sequences.delete(body.sessionId);
            ctx.manager.unregister(body.sessionId);
            ctx.mux.drop(body.sessionId);
            broadcastSessionList(ctx, stream);
            replyOk(stream, env, { ok: true });
            return;
        }
    });

    stream.onClose(() => {
        for (const sid of client.subscriptions) {
            ctx.manager.detachClient(sid, client.clientId);
            ctx.mux.detach(sid, client.clientId);
        }
    });
}

function broadcastSessionList(ctx: DispatchContext, stream: MessageStream): void {
    void stream.send(makeEnv('session-list-update', { sessions: ctx.manager.list() }));
}

function makeEnv(type: Envelope['type'], inner: unknown, replyTo?: string): Envelope {
    return {
        v: 1,
        id: replyTo ?? Math.random().toString(36).slice(2),
        type,
        encrypted: false,
        payload: new TextEncoder().encode(JSON.stringify(inner)),
        timestamp: Date.now()
    };
}

function replyOk(stream: MessageStream, env: Envelope, value: unknown): void {
    void stream.send({ ...makeEnv('rpc-response', { callId: env.id, ok: true, result: value }), id: env.id });
}

function replyErr(stream: MessageStream, env: Envelope, code: string): void {
    void stream.send({ ...makeEnv('rpc-response', { callId: env.id, ok: false, error: { code, message: code } }), id: env.id });
}
```

```typescript
import { RpcCallSchema, RpcResponseSchema } from '@tether/wire';

export async function dispatchRpc(callBytes: Uint8Array, ctx: HandlerContext): Promise<Uint8Array> {
    const call = RpcCallSchema.parse(JSON.parse(new TextDecoder().decode(callBytes)));
    const fn = (handlers as Record<string, (p: unknown, c: HandlerContext) => Promise<unknown>>)[call.method];
    if (!fn) {
        return new TextEncoder().encode(JSON.stringify(RpcResponseSchema.parse({
            callId: call.callId, ok: false,
            error: { code: 'unknown-method', message: call.method }
        })));
    }
    try {
        const result = await fn(call.params, ctx);
        return new TextEncoder().encode(JSON.stringify(RpcResponseSchema.parse({
            callId: call.callId, ok: true, result
        })));
    } catch (err: any) {
        return new TextEncoder().encode(JSON.stringify(RpcResponseSchema.parse({
            callId: call.callId, ok: false,
            error: { code: 'handler-error', message: String(err?.message ?? err) }
        })));
    }
}
```

`spawn` and `kill` need to talk to the running server. Add a minimal HTTP control endpoint at `127.0.0.1:0` in `start.ts` (separate from public host) — a single-purpose admin server bound to localhost only. Actions: POST `/spawn { cwd }` → 200 with sessionId; POST `/kill { sessionId }` → 200.

```typescript
// packages/tether-cli/src/cli/control.ts
import { createServer } from 'node:http';
import { writeFileSync } from 'node:fs';
import type { DispatchContext } from '../transport/sessionDispatch.js';
import type { HomePaths } from '../store/homedir.js';
import { spawnSession } from '../cc/spawn.js';

export async function startControlServer(paths: HomePaths, ctx: DispatchContext): Promise<{ port: number; close: () => Promise<void> }> {
    const server = createServer(async (req, res) => {
        if (req.method !== 'POST') { res.writeHead(405).end(); return; }
        const chunks: Buffer[] = [];
        for await (const c of req) chunks.push(c);
        const body = JSON.parse(Buffer.concat(chunks).toString('utf8') || '{}');

        if (req.url === '/spawn') {
            const r = await spawnSession({ cwd: body.cwd });
            ctx.manager.register({ id: r.sessionId, cwd: body.cwd, sock: r.sock, pid: r.pid, jsonlPath: r.jsonlPath });
            res.writeHead(200, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ sessionId: r.sessionId, sock: r.sock }));
            return;
        }

        if (req.url === '/kill') {
            const rec = ctx.manager.get(body.sessionId);
            if (!rec) { res.writeHead(404).end(); return; }
            try { process.kill(-rec.pid, 'SIGTERM'); } catch {}
            ctx.manager.unregister(body.sessionId);
            res.writeHead(200).end('{}');
            return;
        }

        res.writeHead(404).end();
    });
    return new Promise((resolve) => {
        server.listen(0, '127.0.0.1', () => {
            const addr = server.address();
            if (!addr || typeof addr === 'string') throw new Error('no addr');
            const port = addr.port;
            writeFileSync(paths.root + '/control-port', String(port), 'utf8');
            resolve({ port, close: () => new Promise((r) => server.close(() => r())) });
        });
    });
}
```

```typescript
// packages/tether-cli/src/http/static.ts
import { existsSync, statSync, createReadStream, readFileSync } from 'node:fs';
import { join, resolve, normalize, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import type { IncomingMessage, ServerResponse } from 'node:http';

const __dirname = dirname(fileURLToPath(import.meta.url));
// dist/http/static.js → ../../../tether-app/dist
const APP_DIST = resolve(__dirname, '..', '..', '..', 'tether-app', 'dist');

const MIME: Record<string, string> = {
    '.html': 'text/html; charset=utf-8',
    '.js': 'application/javascript; charset=utf-8',
    '.mjs': 'application/javascript; charset=utf-8',
    '.css': 'text/css; charset=utf-8',
    '.json': 'application/json; charset=utf-8',
    '.svg': 'image/svg+xml',
    '.png': 'image/png',
    '.ico': 'image/x-icon',
    '.webmanifest': 'application/manifest+json'
};

export function staticHandler(req: IncomingMessage, res: ServerResponse): void {
    if (!req.url || req.method !== 'GET') {
        res.writeHead(404).end();
        return;
    }
    if (req.url === '/wt' || req.url.startsWith('/wt?')) {
        // WebSocket upgrade handled elsewhere, never reach here
        return;
    }

    const urlPath = req.url.split('?')[0] === '/' ? '/index.html' : req.url.split('?')[0];
    const safePath = normalize(urlPath).replace(/^\/+/, '');
    const fullPath = resolve(APP_DIST, safePath);

    if (!fullPath.startsWith(APP_DIST)) { res.writeHead(403).end(); return; }
    if (!existsSync(fullPath) || !statSync(fullPath).isFile()) {
        // SPA fallback to index.html
        const fallback = join(APP_DIST, 'index.html');
        if (!existsSync(fallback)) {
            res.writeHead(503, { 'Content-Type': 'text/plain' });
            res.end('PWA not built — run `pnpm -C packages/tether-app build`');
            return;
        }
        res.writeHead(200, { 'Content-Type': 'text/html; charset=utf-8' });
        res.end(readFileSync(fallback));
        return;
    }
    const ext = fullPath.slice(fullPath.lastIndexOf('.'));
    res.writeHead(200, { 'Content-Type': MIME[ext] ?? 'application/octet-stream' });
    createReadStream(fullPath).pipe(res);
}
```

```typescript
// packages/tether-app/src/transport/webtransport.ts
import { encodeEnvelope, decodeEnvelope, type Envelope } from '@tether/wire';
import type { ClientStream } from './client.js';

export async function connectWebTransport(url: string): Promise<ClientStream> {
    const wt = new WebTransport(url);
    await wt.ready;

    let onMsg: ((e: Envelope) => void) | undefined;
    let onClose: (() => void) | undefined;
    wt.closed.then(() => onClose?.()).catch(() => onClose?.());

    (async () => {
        const reader = wt.incomingBidirectionalStreams.getReader();
        for (;;) {
            const { value: bidi, done } = await reader.read();
            if (done) break;
            (async () => {
                const r = bidi.readable.getReader();
                const chunks: Uint8Array[] = [];
                for (;;) {
                    const { value, done } = await r.read();
                    if (done) break;
                    chunks.push(value);
                }
                const total = chunks.reduce((n, c) => n + c.length, 0);
                const buf = new Uint8Array(total);
                let off = 0;
                for (const c of chunks) { buf.set(c, off); off += c.length; }
                onMsg?.(decodeEnvelope(buf));
            })();
        }
    })();

    return {
        kind: 'wt',
        send: async (env) => {
            const bidi = await wt.createBidirectionalStream();
            const w = bidi.writable.getWriter();
            await w.write(encodeEnvelope(env));
            await w.close();
        },
        onMessage: (h) => { onMsg = h; },
        onClose: (h) => { onClose = h; },
        close: () => wt.close()
    };
}
```

```typescript
// packages/tether-app/src/crypto/client.ts
import sodium from 'libsodium-wrappers';

let ready: Promise<void> | undefined;
export async function ensureSodiumReady(): Promise<typeof sodium> {
    if (!ready) ready = sodium.ready;
    await ready;
    return sodium;
}

export interface ClientKeypair { publicKey: Uint8Array; secretSeed: Uint8Array; }

export async function generateClientKeypair(): Promise<ClientKeypair> {
    const s = await ensureSodiumReady();
    const seed = s.randombytes_buf(32);
    const kp = s.crypto_box_seed_keypair(seed);
    return { publicKey: kp.publicKey, secretSeed: seed };
}

export async function deriveSharedSecret(serverPub: Uint8Array, clientSeed: Uint8Array): Promise<Uint8Array> {
    const s = await ensureSodiumReady();
    const kp = s.crypto_box_seed_keypair(clientSeed);
    return s.crypto_scalarmult(kp.privateKey, serverPub);
}

export async function aesGcmEncrypt(key: Uint8Array, plaintext: Uint8Array): Promise<Uint8Array> {
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const cryptoKey = await crypto.subtle.importKey('raw', key, 'AES-GCM', false, ['encrypt']);
    const ct = new Uint8Array(await crypto.subtle.encrypt({ name: 'AES-GCM', iv }, cryptoKey, plaintext));
    const out = new Uint8Array(1 + 12 + ct.length);
    out[0] = 0; // version
    out.set(iv, 1);
    out.set(ct, 13);
    return out;
}

export async function aesGcmDecrypt(key: Uint8Array, bundle: Uint8Array): Promise<Uint8Array | null> {
    if (bundle[0] !== 0) return null;
    const iv = bundle.slice(1, 13);
    const ct = bundle.slice(13);
    try {
        const cryptoKey = await crypto.subtle.importKey('raw', key, 'AES-GCM', false, ['decrypt']);
        return new Uint8Array(await crypto.subtle.decrypt({ name: 'AES-GCM', iv }, cryptoKey, ct));
    } catch { return null; }
}
```

```typescript
import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { connect } from '../transport/client.js';
import { generateClientKeypair, ensureSodiumReady } from '../crypto/client.js';
import { saveAuth } from '../store/auth.js';
import { encodeEnvelope, decodeEnvelope, PairResponseSchema } from '@tether/wire';

export function Pair() {
    const nav = useNavigate();
    const [url, setUrl] = useState('');
    const [busy, setBusy] = useState(false);
    const [error, setError] = useState<string | undefined>();

    async function pair() {
        setBusy(true);
        setError(undefined);
        try {
            const m = url.match(/^tether:\/\/([^?]+)\?pubkey=([^&]+)&token=(.+)$/);
            if (!m) throw new Error('Pasted URL is not a valid tether:// pairing link.');
            const [, host, serverPubKeyB64, oneTimeToken] = m;
            const serverHttpsUrl = `https://${host}`;
            const stream = await connect(serverHttpsUrl);
            await ensureSodiumReady();
            const kp = await generateClientKeypair();

            const reply = await new Promise<any>((resolve, reject) => {
                const tid = setTimeout(() => reject(new Error('pair timeout')), 10_000);
                stream.onMessage((env) => {
                    if (env.type !== 'pair-response') return;
                    clearTimeout(tid);
                    const body = JSON.parse(new TextDecoder().decode(env.payload));
                    if (body.error) return reject(new Error(body.error.message));
                    resolve(PairResponseSchema.parse(body));
                });
                void stream.send({
                    v: 1, id: 'pair', type: 'pair-request', encrypted: false,
                    payload: new TextEncoder().encode(JSON.stringify({
                        clientPubKey: btoa(String.fromCharCode(...kp.publicKey)),
                        oneTimeToken: decodeURIComponent(oneTimeToken)
                    })),
                    timestamp: Date.now()
                });
            });

            await saveAuth({
                serverUrl: serverHttpsUrl,
                serverPubKey: serverPubKeyB64,
                bearerToken: reply.bearerToken,
                expiresAt: reply.expiresAt,
                machineId: reply.machineId,
                clientPubKey: btoa(String.fromCharCode(...kp.publicKey)),
                clientSeed: btoa(String.fromCharCode(...kp.secretSeed))
            });
            nav('/');
        } catch (err: any) {
            setError(err?.message ?? String(err));
        } finally {
            setBusy(false);
        }
    }

    return (
        <main style={{ padding: '1rem' }}>
            <h2>Pair with your Tether server</h2>
            <p>On your server, run <code>tether start</code> and paste the <code>tether://...</code> URL it prints.</p>
            <textarea value={url} onChange={(e) => setUrl(e.target.value)} rows={3} style={{ width: '100%' }} />
            <button onClick={pair} disabled={busy || !url}>Pair</button>
            {error && <p style={{ color: 'red' }}>{error}</p>}
        </main>
    );
}
```

```typescript
// packages/tether-app/src/store/sessions.ts
export interface SessionView {
    id: string;
    cwd: string;
    createdAt: number;
    attachedClients: number;
    attachedTerminal: boolean;
    lastActivity?: number;
}
export interface ChatLine { seq: number; line: unknown; }

export interface StoreState {
    sessions: SessionView[];
    chats: Map<string, ChatLine[]>;
    activeSessionId?: string;
}

export interface SessionStore {
    snapshot: () => StoreState;
    subscribe: (l: () => void) => () => void;
    applyListUpdate: (s: SessionView[]) => void;
    applyChatLine: (sessionId: string, seq: number, line: unknown) => void;
    setActiveSession: (id: string | undefined) => void;
}

export function createSessionStore(): SessionStore {
    let state: StoreState = { sessions: [], chats: new Map() };
    const listeners = new Set<() => void>();
    const notify = () => { for (const l of listeners) l(); };
    return {
        snapshot: () => state,
        subscribe: (l) => { listeners.add(l); return () => listeners.delete(l); },
        applyListUpdate: (s) => {
            state = { ...state, sessions: s };
            notify();
        },
        applyChatLine: (id, seq, line) => {
            const log = state.chats.get(id) ?? [];
            const next = new Map(state.chats);
            next.set(id, [...log, { seq, line }]);
            state = { ...state, chats: next };
            notify();
        },
        setActiveSession: (id) => {
            state = { ...state, activeSessionId: id };
            notify();
        }
    };
}
```

```typescript
// packages/tether-app/src/transport/manager.ts
import { useEffect, useState } from 'react';
import { type Envelope, RpcResponseSchema } from '@tether/wire';
import { connect, type ClientStream } from './client.js';
import { loadAuth } from '../store/auth.js';
import { createSessionStore, type SessionStore, type SessionView } from '../store/sessions.js';

let stream: ClientStream | undefined;
let connecting: Promise<ClientStream> | undefined;
export const sessionStore: SessionStore = createSessionStore();
const pending = new Map<string, (resp: any) => void>();

export async function ensureConnected(): Promise<ClientStream> {
    if (stream) return stream;
    if (connecting) return connecting;
    const auth = await loadAuth();
    if (!auth) throw new Error('not paired');
    connecting = (async () => {
        const s = await connect(auth.serverUrl);
        s.onMessage((env) => routeIncoming(env));
        s.onClose(() => { stream = undefined; });
        stream = s;
        connecting = undefined;
        return s;
    })();
    return connecting;
}

function routeIncoming(env: Envelope) {
    if (env.type === 'session-list-update') {
        const body = JSON.parse(new TextDecoder().decode(env.payload)) as { sessions: SessionView[] };
        sessionStore.applyListUpdate(body.sessions);
        return;
    }
    if (env.type === 'session-update') {
        const body = JSON.parse(new TextDecoder().decode(env.payload)) as { sessionId: string; seq: number; line: unknown };
        sessionStore.applyChatLine(body.sessionId, body.seq, body.line);
        return;
    }
    if (env.type === 'rpc-response') {
        const body = RpcResponseSchema.parse(JSON.parse(new TextDecoder().decode(env.payload)));
        pending.get(body.callId)?.(body);
        pending.delete(body.callId);
        return;
    }
}

export async function rpcCall<T>(method: string, params: unknown): Promise<T> {
    const s = await ensureConnected();
    const callId = Math.random().toString(36).slice(2);
    const call = { method, params, callId };
    const env: Envelope = {
        v: 1, id: callId, type: 'rpc-call', encrypted: false,
        payload: new TextEncoder().encode(JSON.stringify(call)),
        timestamp: Date.now()
    };
    await s.send(env);
    return new Promise<T>((resolve, reject) => {
        pending.set(callId, (r) => r.ok ? resolve(r.result as T) : reject(new Error(r.error?.message ?? 'rpc-error')));
    });
}

export async function sendSimple(type: Envelope['type'], inner: unknown): Promise<void> {
    const s = await ensureConnected();
    await s.send({
        v: 1, id: Math.random().toString(36).slice(2), type, encrypted: false,
        payload: new TextEncoder().encode(JSON.stringify(inner)), timestamp: Date.now()
    });
}

export function useStoreSnapshot() {
    const [snap, setSnap] = useState(sessionStore.snapshot());
    useEffect(() => sessionStore.subscribe(() => setSnap(sessionStore.snapshot())), []);
    return snap;
}
```

```typescript
import webpush from 'web-push';
import type { HomePaths } from '../store/homedir.js';

export interface VapidKeys { publicKey: string; privateKey: string; subject: string; }

export function ensureVapidKeys(paths: HomePaths, subject: string): VapidKeys {
    if (existsSync(paths.vapid)) {
```

```typescript
// packages/tether-cli/src/push/subscriptions.ts
import { existsSync, readFileSync, writeFileSync } from 'node:fs';
import type { HomePaths } from '../store/homedir.js';

export interface Subscription {
    endpoint: string;
    keys: { p256dh: string; auth: string };
    eventTypes: ('permission' | 'completion' | 'error' | 'plan-mode')[];
    bearerToken: string;     // ties subscription to a paired client
    registeredAt: number;
}

export class SubscriptionStore {
    private subs: Subscription[];
    constructor(private paths: HomePaths) {
        this.subs = existsSync(paths.subscriptions)
            ? JSON.parse(readFileSync(paths.subscriptions, 'utf8'))
            : [];
    }
    private save() { writeFileSync(this.paths.subscriptions, JSON.stringify(this.subs, null, 2), 'utf8'); }

    upsert(sub: Subscription) {
        this.subs = this.subs.filter((s) => s.endpoint !== sub.endpoint);
        this.subs.push(sub);
        this.save();
    }
    remove(endpoint: string) {
        this.subs = this.subs.filter((s) => s.endpoint !== endpoint);
        this.save();
    }
    listForEvent(event: Subscription['eventTypes'][number]): Subscription[] {
        return this.subs.filter((s) => s.eventTypes.includes(event));
    }
    all(): Subscription[] { return [...this.subs]; }
}
```

```typescript
// packages/tether-cli/src/push/triggers.ts
export type TriggerKind = 'permission' | 'completion' | 'error' | 'plan-mode';

export interface TriggerEvent {
    sessionId: string;
    kind: TriggerKind;
    summary: string;
    at: number;
}

export interface TriggerWatcherOptions { idleMs?: number; }

export class TriggerWatcher {
    private pendingCompletion = new Map<string, ReturnType<typeof setTimeout>>();
    constructor(
        private emit: (ev: TriggerEvent) => void,
        private opts: TriggerWatcherOptions = {}
    ) {}

    observe(sessionId: string, line: any): void {
        if (!line || typeof line !== 'object') return;

        if (line.type === 'tool_use' && line.permissionMode === 'ask') {
            this.emit({ sessionId, kind: 'permission', summary: `Permission for ${line.name}`, at: Date.now() });
            return;
        }
        if (line.type === 'error') {
            this.emit({ sessionId, kind: 'error', summary: String(line.message ?? 'error'), at: Date.now() });
            return;
        }
        if (line.mode === 'plan') {
            this.emit({ sessionId, kind: 'plan-mode', summary: 'Entered plan mode', at: Date.now() });
            return;
        }
        if ((line.type === 'message' || line.role === 'assistant') && line.stop_reason === 'end_turn') {
            const t = this.pendingCompletion.get(sessionId);
            if (t) clearTimeout(t);
            const idle = this.opts.idleMs ?? 2000;
            const handle = setTimeout(() => {
                this.emit({ sessionId, kind: 'completion', summary: 'Agent finished', at: Date.now() });
                this.pendingCompletion.delete(sessionId);
            }, idle);
            this.pendingCompletion.set(sessionId, handle);
            return;
        }

        // Any non-completion line cancels a pending completion
        const t = this.pendingCompletion.get(sessionId);
        if (t) { clearTimeout(t); this.pendingCompletion.delete(sessionId); }
    }
}
```

```typescript
function urlBase64ToUint8Array(base64: string): Uint8Array {
    const padding = '='.repeat((4 - base64.length % 4) % 4);
    const base = (base64 + padding).replace(/-/g, '+').replace(/_/g, '/');
    const raw = atob(base);
    const out = new Uint8Array(raw.length);
    for (let i = 0; i < raw.length; i++) out[i] = raw.charCodeAt(i);
    return out;
}

export async function ensurePushSubscription(eventTypes: ('permission' | 'completion' | 'error' | 'plan-mode')[]): Promise<void> {
    if (!('serviceWorker' in navigator) || !('PushManager' in window)) {
        throw new Error('Push not supported in this browser');
    }
    const reg = await navigator.serviceWorker.ready;
    const perm = await Notification.requestPermission();
    if (perm !== 'granted') throw new Error('Notification permission denied');

    let sub = await reg.pushManager.getSubscription();

    // First do the RPC with no key to learn server's public key
    if (!sub) {
        // Need server's vapid public key to subscribe. We do a tentative RPC with a marker key, server returns vapidPublicKey.
        const handshake = await rpcCall<{ vapidPublicKey: string }>('registerPushSubscription', {
            endpoint: 'https://placeholder.invalid/',
            keys: { p256dh: 'AA', auth: 'AA' },
            eventTypes
        });
        sub = await reg.pushManager.subscribe({
            userVisibleOnly: true,
            applicationServerKey: urlBase64ToUint8Array(handshake.vapidPublicKey)
        });
    }

    const json = sub.toJSON();
    await rpcCall('registerPushSubscription', {
        endpoint: json.endpoint!,
        keys: { p256dh: json.keys!.p256dh, auth: json.keys!.auth },
        eventTypes
    });
```

```typescript
import { ensureSodiumReady } from './client.js';

export async function deriveClientSharedAndDataKey(serverPubB64: string, clientSeedB64: string): Promise<Uint8Array> {
    const s = await ensureSodiumReady();
    const seed = Uint8Array.from(atob(clientSeedB64), (c) => c.charCodeAt(0));
    const serverPub = Uint8Array.from(atob(serverPubB64), (c) => c.charCodeAt(0));

    const shared = s.crypto_scalarmult(kp.privateKey, serverPub);
    // HKDF
    const salt = kp.publicKey;
    return await hkdf(shared, salt, 'tether-data-v1', 32);

    const bits = await crypto.subtle.deriveBits({
        name: 'HKDF', hash: 'SHA-256', salt, info: new TextEncoder().encode(info)
```

```typescript
// packages/tether-cli/src/session/recover.ts
import { readdirSync, statSync } from 'node:fs';
import { join } from 'node:path';
import { readFileSync, existsSync } from 'node:fs';
import type { HomePaths } from '../store/homedir.js';

export interface RecoveredSession {
    id: string;
    sock: string;
    cwd: string;
    jsonlPath: string;
}

export function recoverSessions(paths: HomePaths): RecoveredSession[] {
    const stateFile = join(paths.root, 'state.json');
    if (!existsSync(stateFile)) return [];
    const state = JSON.parse(readFileSync(stateFile, 'utf8'));
    const out: RecoveredSession[] = [];
    for (const s of state.sessions ?? []) {
        if (existsSync(s.sock) && statSync(s.sock).isSocket()) {
            out.push({ id: s.id, sock: s.sock, cwd: s.cwd, jsonlPath: s.jsonlPath });
        }
    }
    return out;
}
```
